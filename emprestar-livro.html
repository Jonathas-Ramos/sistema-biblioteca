package br.com.biblioteca.model; import org.springframework.data.annotation.Id; import org.springframework.data.mongodb.core.mapping.Document; @Document(collection = "livros") public class Livro { @Id private String id; private String titulo; private int quantidadeDisponivel; // Getters e Setters public String getId() { return id; } public void setId(String id) { this.id = id; } public String getTitulo() { return titulo; } public void setTitulo(String titulo) { this.titulo = titulo; } public int getQuantidadeDisponivel() { return quantidadeDisponivel; } public void setQuantidadeDisponivel(int quantidadeDisponivel) { this.quantidadeDisponivel = quantidadeDisponivel; } }
